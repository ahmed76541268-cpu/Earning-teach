    <div class="button-grid">
      <button class="game-btn" onclick="playAd('Spin to Win', 10)">🎡 Spin</button>
      <button class="game-btn" onclick="playAd('Watch Video', 20)">🎥 Watch</button>
      <button class="game-btn" onclick="playAd('Claim Reward', 50)">💎 Claim</button>

  <!-- AuruAds Script -->
  <script src="https://auruads.com/cdn/ads.js?app_uid=3084"></script>

  <script>
    let coins = 0;

    // Load saved coins
    window.onload = function() {
      let savedCoins = localStorage.getItem("coins");
      if (savedCoins) {
        coins = parseInt(savedCoins);
        document.getElementById("coinBalance").innerText = coins;
      }
    };

    function playAd(title, reward) {
      // Call AuruAds
      window.showAuruads()
        .then(() => {
          // ✅ Grant reward after ad success
          coins += reward;
          document.getElementById("coinBalance").innerText = coins;
          localStorage.setItem("coins", coins);

          alert("✅ " + title + " completed! You earned +" + reward + " coins.");
        })
        .catch(e => {
          // ❌ Handle errors
          console.error("Ad failed:", e);
          alert("⚠️ Ad failed to load. Please try again.");
        });
    }
  </script>
